## 挂载系统安装文件

{% codeblock lang:bash%}
mount /dev/cdrom /mnt
{% endcodeblock %}

## 进入SERVER

{% codeblock lang:bash%}
cd /mnt/Server/
{% endcodeblock %}

<!-- more -->

## 查找gcc

{% codeblock lang:bash%}
rpm -ivh gcc*
{% endcodeblock %}

## 安装依赖

{% codeblock lang:bash%}
rpm -ivh kernel-headers-2.6.18-194.el5.i386.rpm
rpm -ivh glibc-headers-2.5-49.i386.rpm
rpm -ivh glibc-devel-2.5-49.i386.rpm
rpm -ivh libgomp-4.4.0-6.el5.i386.rpm
{% endcodeblock %}

## 安装

{% codeblock lang:bash%}
rpm -ivh gcc-4.1.2-48.el5.i386.rpm
{% endcodeblock %}

## 查看是否安装成功
{% codeblock lang:bash%}
gcc -v
{% endcodeblock %}
