## 创建文件，加入如下内容

{% codeblock lang:bash %}
cd ~
touch .git-credentials
vi .git-credentials
https://{username}:{password}@{domain}{:port}
{% endcodeblock %}

## 执行如下命令

{% codeblock lang:bash %}
git config --global credential.helper store
{% endcodeblock %}

## 打开~/.gitconfig文件查看

{% codeblock lang:txt %}
[credential]
helper = store
{% endcodeblock %}
